	title = rsch.1.t
	desc = rsch.1.d
	picture = GFX_dynrsch

		NOT = { has_country_flag = dynamic_research_advanced_1 }

		}

		hours = 1
	}
	
	option = {
		name = rsch.1.a
		add_research_slot = 1
		set_country_flag = dynamic_research_advanced_1
	}
}
