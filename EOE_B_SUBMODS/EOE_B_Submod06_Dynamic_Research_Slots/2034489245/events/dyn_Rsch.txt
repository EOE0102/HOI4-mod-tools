	title = rsch.1.t
	desc = rsch.1.d
	picture = GFX_dynrsch

	fire_only_once = no
	
	trigger = {
		NOT = { has_country_flag = research_slot_1 }

		hours = 1
	}
	
	option = {
		name = rsch.1.a
		add_research_slot = 1
		set_country_flag = research_slot_1
	}
}
